def is_Nguyen_To(n):
    if(n ==2 or n == 1): return True
    if(n<=0): return False
    count = 0
    for i in range(1, int(n/2)+1):
        if n % i ==0:
            count+=1
    if(count > 1):
        return False
    
    return True

so_tu_nhien = 35
so = so_tu_nhien
lst = []

def tim_thua_so(num):
    x = 2
    while True:
        if(is_Nguyen_To(x)) & (num%x==0):
            return x
        x+=1

while True:
    if not is_Nguyen_To((int(so_tu_nhien))):
        thua_so = tim_thua_so(so_tu_nhien)
        lst.append(str(thua_so))
        so_tu_nhien = so_tu_nhien / thua_so

def is_Nguyen_To(n):
    if(n ==2 or n == 1): return True
    if(n<=0): return False
    count = 0
    for i in range(1, int(n/2)+1):
        if n % i ==0:
            count+=1
    if(count > 1):
        return False
    
    return True

so_tu_nhien = 36
so = so_tu_nhien
lst = []

def tim_thua_so(num):
    x = 2
    while True:
        if(is_Nguyen_To(x)) & (num%x==0):
            return x
        x+=1

while True:
    if not is_Nguyen_To((int(so_tu_nhien))):
        thua_so = tim_thua_so(so_tu_nhien)
        lst.append(str(thua_so))
        so_tu_nhien = so_tu_nhien / thua_so
    else:
        lst.append(str(int(so_tu_nhien)))
        break
    
print(str(so) + " = "+ " x ".join(lst))
print(f"Tổng các chữ số {so} là : {sum([int(i) for i in str(so)])}")
